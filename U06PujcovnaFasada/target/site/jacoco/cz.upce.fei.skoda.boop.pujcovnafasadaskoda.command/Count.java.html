package cz.upce.fei.skoda.boop.pujcovnafasadaskoda.command;

import cz.upce.fei.skoda.boop.pujcovnafasadaskoda.spravce.IManager;

/**
 * Class representing count command
 * @author Jiri Skoda <jiri.skoda@student.upce.cz>
 */
public class Count extends Command
{
    /**
     * Creates new handler for count command
     * @param dataManager Manager of data
     */
    public Count(IManager dataManager)
    {
        super(dataManager);
        this.aliases = new String[]{"count", "pocet", "c"};
    }

    @Override
    public String getHelp()
    {
        return "Vypise pocet prvku v seznamu";
    }

    @Override
    public String execute(String[] arguments)
    {
        this.error = false;
        return String.valueOf(this.dataManager.countElements());
    }
    
}
